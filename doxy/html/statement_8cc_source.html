252  if (parameters.size() == 0) {

254  }

260  Parameters::const_iterator end = parameters.end();
261 
262  for (; it < end; ++it) {
263  Values::Field* field = *it;
264 
265  if (field != NULL) {
266  int pos;
267  if (field->index > 0) {
268  pos = field->index;
269  }

272  }

276  status = sqlite3_bind_int(_handle, pos,

278  } break;
279  case SQLITE_FLOAT: {
280  status = sqlite3_bind_double(_handle, pos,
281  ((Values::Float*)field)->value);
282  } break;
283  case SQLITE_TEXT: {
284  status = sqlite3_bind_text(_handle, pos,
285  ((Values::Text*)field)->value.c_str(),
286  ((Values::Text*)field)->value.size(), SQLITE_TRANSIENT);

289  status = sqlite3_bind_blob(_handle, pos,
290  ((Values::Blob*)field)->value,
291  ((Values::Blob*)field)->length, SQLITE_TRANSIENT);
292  } break;
293  case SQLITE_NULL: {
294  status = sqlite3_bind_null(_handle, pos);
295  } break;
296  }

787 void Statement::GetRow(Row* row, sqlite3_stmt* stmt) {
788  int rows = sqlite3_column_count(stmt);
789 
790  for (int i = 0; i < rows; i++) {
791  int type = sqlite3_column_type(stmt, i);
792  const char* name = sqlite3_column_name(stmt, i);
793  switch (type) {
794  case SQLITE_INTEGER: {
795  row->push_back(new Values::Integer(name, sqlite3_column_int64(stmt, i)));
796  } break;
797  case SQLITE_FLOAT: {
798  row->push_back(new Values::Float(name, sqlite3_column_double(stmt, i)));
799  } break;
800  case SQLITE_TEXT: {
801  const char* text = (const char*)sqlite3_column_text(stmt, i);
802  int length = sqlite3_column_bytes(stmt, i);
803  row->push_back(new Values::Text(name, length, text));
804  } break;
805  case SQLITE_BLOB: {
806  const void* blob = sqlite3_column_blob(stmt, i);
807  int length = sqlite3_column_bytes(stmt, i);
808  row->push_back(new Values::Blob(name, length, blob));
809  } break;
810  case SQLITE_NULL: {
811  row->push_back(new Values::Null(name));
812  } break;
813  default:
814  assert(false);
815  }
816  }

854 void Statement::CleanQueue() {
855  Nan::HandleScope scope;
856 
857  if (prepared && !queue.empty()) {
858  // This statement has already been prepared and is now finalized.
859  // Fire error for all remaining items in the queue.
860  EXCEPTION("Statement is already finalized", SQLITE_MISUSE, exception);
861  Local<Value> argv[] = { exception };
862  bool called = false;
863 
864  // Clear out the queue so that this object can get GC'ed.
865  while (!queue.empty()) {
866  Call* call = queue.front();
867  queue.pop();
868 
869  Local<Function> cb = Nan::New(call->baton->callback);
870 
871  if (prepared && !cb.IsEmpty() &&
872  cb->IsFunction()) {
873  TRY_CATCH_CALL(handle(), cb, 1, argv);
874  called = true;
875  }
876 
877  // We don't call the actual callback, so we have to make sure that
878  // the baton gets destroyed.
879  delete call->baton;
880  delete call;
881  }
882 
883  // When we couldn't call a callback function, emit an error on the
884  // Statement object.
885  if (!called) {
886  Local<Value> info[] = { Nan::New("error").ToLocalChecked(), exception };
887  EMIT_EVENT(handle(), 2, info);
888  }
889  }
890  else while (!queue.empty()) {
891  // Just delete all items in the queue; we already fired an event when
892  // preparing the statement failed.
893  Call* call = queue.front();
894  queue.pop();
895 
896  // We don't call the actual callback, so we have to make sure that
897  // the baton gets destroyed.
898  delete call->baton;
899  delete call;
900  }
901 }
